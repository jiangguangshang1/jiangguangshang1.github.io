C++最小公倍数代码：

#include<iostream>
using namespace std;

//求最大公约数
int gcb(int a, int b)
{
    int c = 0;
    while(c = a % b)
    {
        a = b;
        b = c;
    }
    return b;
}

int main()
{
    int a, b;
    while(cin >> a >> b)
    {
        //最小公倍数 = 两数之积除以最大公约数
        cout << a * b / gcb(a, b) << endl;
    }
    return 0;
}
